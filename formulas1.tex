\noindent\textbf{Formulas}
 
\noindent\textsl{discrete time growth:}
\begin{itemize}
\item $N_T = N_0 \lambda^T$
\item $\lambda = f + p$
\item $\R = f/(1-p)$
\end{itemize}
 
\noindent\textsl{continuous time growth:}
\begin{itemize}
\item $N(t) = N(0) \exp(rt)$
\item $r=b-d$
\item $\R = b/d$
\end{itemize}
